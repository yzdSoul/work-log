**关键代码变更：**
```python
# 懒加载 MongoDB 连接
_client = None
_db = None

def _ensure_collections():
    global _client, _db
    if _client is None:
        MONGO_URI = os.environ.get("MONGO_URI") or os.environ.get("MONGODB_URI")
        _client = MongoClient(MONGO_URI, tlsCAFile=certifi.where())
        _db = _client["daily_motto"]
    return _db["quotes"], _db["pending_quotes"], _db["visit_logs"]
```

- 所有 MongoDB 操作添加 `try/except` 保护，避免 500 错误
